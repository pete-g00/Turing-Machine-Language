alphabet = {0, 1}
module isDiv2Rec {
    // recursive case: not at the end => move closer to the end
    if 0, 1 {
        move right
        goto isDiv2Rec
    } 
    // base case: at the end => check final letter 0
    if blank {
        move left
        if 0 {
            accept
        } if 1, blank {
            reject
        }
    }
}
